27inline std::vector<std::array<double, 2>> CreateTrapezoidalQuadrature(std::span<const double> grid) {
28 const auto n{grid.size()};
29 const auto [grid_min, grid_max] = std::ranges::minmax(grid);
30 const auto grid_range{grid_max - grid_min};
31 auto quadrature = std::vector<std::array<double, 2>>{
32 {-1., (grid[1] - grid[0]) / grid_range},
33 };
34 std::ranges::transform(
35 std::views::iota(1U, n - 1), std::back_inserter(quadrature),
36 [grid, gm = grid_min, grid_range](auto i) {
37 return std::array{
38 2. * (grid[i] - gm) / grid_range - 1., (grid[i + 1] - grid[i - 1]) / grid_range
39 };
40 }
41 );
42 quadrature.push_back({1., (grid[n - 1] - grid[n - 2]) / grid_range});
43 return quadrature;
44}

56inline std::vector<std::array<double, 2>> CreateGaussLegendreQuadrature(const size_t order) {
57 constexpr auto max_iterations = 1000U;
58 constexpr auto convergence_tolerance = std::numeric_limits<double>::epsilon();
59
60 // If order is 0, return the base case
61 if (order < 1) {
62 return {{0., 2.}}; // point -> 0, weight->2
63 }
64
65 const size_t n_points{order}; // GL has n points for order n
66 const size_t n_unique{(n_points + 1) / 2}; // Unique points due to symmetry
67 std::vector<double> points(n_points, 0.); // Quadrature points
68 std::vector<double> weights(n_points, 0.); // Quadrature weights
69
70 // Find Gauss-Legendre (GL) points using Newton-Raphson method
71 for (auto i_GL_point : std::views::iota(0U, n_unique)) {
72 // Initial guess
73 auto x_it = std::cos(
74 std::numbers::pi * (static_cast<double>(i_GL_point) + 0.75) /
75 (static_cast<double>(n_points) + 0.5)
76 );
77
78 // Newton-Raphson iteration to find root of P_n(x)
79 bool converged = false;
80 for ([[maybe_unused]] auto iteration : std::views::iota(0U, max_iterations)) {
81 const auto x_old = x_it; // Store old value for convergence check
82
83 // Evaluate Legendre polynomials P_{n-1}(x) and P_n(x)
84 const auto p_n_minus_1 = kynema::math::LegendrePolynomial(n_points - 1, x_it);
85 const auto p_n = kynema::math::LegendrePolynomial(n_points, x_it);
86
87 // Compute derivative P'_n(x) using recurrence relation
88 const auto p_n_prime =
89 static_cast<double>(n_points) * (x_it * p_n - p_n_minus_1) / (x_it * x_it - 1.);
90
91 // Newton update: x_{n+1} = x_n - f(x_n)/f'(x_n)
92 x_it -= p_n / p_n_prime;
93
94 // Check convergence
95 if (std::abs(x_it - x_old) <= convergence_tolerance) {
96 converged = true;
97 break;
98 }
99 }
100 if (!converged) {
101 throw std::runtime_error(
102 "Newton-Raphson failed to converge for GL point " + std::to_string(i_GL_point) +
103 " of order " + std::to_string(n_points)
104 );
105 }
106
107 // Store symmetric points
108 points[i_GL_point] = -x_it; // left side
109 points[n_points - 1 - i_GL_point] = x_it; // right side
110
111 // Compute GL weights: w = 2 / ((1 - x²) * [P'_n(x)]²)
112 const auto p_n_minus_1 = kynema::math::LegendrePolynomial(n_points - 1, x_it);
113 const auto p_n = kynema::math::LegendrePolynomial(n_points, x_it);
114 const auto p_n_prime =
115 static_cast<double>(n_points) * (x_it * p_n - p_n_minus_1) / (x_it * x_it - 1.);
116 const auto weight = 2. / ((1. - x_it * x_it) * p_n_prime * p_n_prime);
117
118 // Store symmetric weights
119 weights[i_GL_point] = weight;
120 weights[n_points - 1 - i_GL_point] = weight;
121 }
122
123 // Build and sort quadrature pairs
124 auto quadrature = std::vector<std::array<double, 2>>(n_points);
125 for (auto i_GL_point : std::views::iota(0U, n_points)) {
126 quadrature[i_GL_point] = {points[i_GL_point], weights[i_GL_point]};
127 }
128 std::ranges::sort(quadrature, [](const auto& a, const auto& b) {
129 return a[0] < b[0];
130 });
131 return quadrature;
132}
